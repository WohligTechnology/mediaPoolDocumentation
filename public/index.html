Schema Definitions

User: object

_id: string (objectId)
accessToken: string
name: string
profilePic: string
email: string (email)
accessLevel: string
onesignal: string
lastLogin: string (date-time)
boardCount: number
basket: string[]

Array of Media ObjectId's

Example
{
  "_id": "507f1f77bcf86cd799439011",
  "accessToken": 829739238729,
  "name": "chintan",
  "profilePic": "http://mediapool.io:4400/images/tui.png",
  "email": "chintan.shah@sptr.co",
  "accessLevel": "Admin",
  "onesignal": "507f1f77bcf86cd799439011",
  "lastLogin": "2016-10-05T14:35:33.593Z",
  "boardCount": 102,
  "basket": [
    "507f1f77bcf86cd799439011"
  ]
}

Media: object

_id: number (objectId)
name: string
classifications: string[]

Array of Classifications

fields: object[]
url: string
mimeType: string
mediaType: string
keyword: string[]
files: object[]
Example
{
  "_id": "507f1f77bcf86cd799439011",
  "name": "Robinson Club Landskron (AC1813116)",
  "classifications": [
    "/Tags/Product/TUI Chain/ROBINSON",
    "/Tags/Product/Market Concept/ROBINSON"
  ],
  "fields": {
    "imageWidth": 6144,
    "imageHeight": 4096
  },
  "url": "robinson-club-landskron",
  "mimeType": "image/jpeg",
  "mediaType": "Image",
  "keyword": [
    "Accommodations",
    "General facilities"
  ],
  "files": {
    "url": "http://cdn.content.tuigroup.com/tuidamdevs3/2017_6/30_10/937cee9e-dfcd-4d34-81ff-a7a200a76806/lk58_4824.jpg",

  }
}
